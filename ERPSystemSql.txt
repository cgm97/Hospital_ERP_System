---- 재고 ITEM 테이블 ----
CREATE TABLE ITEM(
    CODE NUMBER(6) NOT NULL primary key,
    NAME NVARCHAR2(50) NOT NULL,
    TYPE NVARCHAR2(50) NOT NULL,
    DEPARTMENT NVARCHAR2(20) NOT NULL,
    totalcount number(38)
    );
